CREATE TABLE CONV_ACCOUNTS
(
  DISCRIMINATOR               CHAR(1 CHAR),
  NAME                        VARCHAR2(255 CHAR),
  ACCNO                       VARCHAR2(255 CHAR),
  OPENDATE                    TIMESTAMP(6),
  STATUS                      NUMBER(10),
  PRODUCT                     NUMBER(19,2),
  PROFITRATE                  NUMBER(19,2),
  AUTORENEW                   NUMBER(1),
  EXPIRYDATE                  TIMESTAMP(6),
  PRINCIPAL                   NUMBER(19,2),
  PROFIT                      NUMBER(19,2),
  PRODUCT_CODE                NUMBER(19),
  BANK_SWIFT_CODE             VARCHAR2(10 BYTE),
  BRANCH_ROUTING_NUMBER       NUMBER(19),
  CURRENCY_CODE               VARCHAR2(10 BYTE),
  OWNER_INSTITUTE_BIC         VARCHAR2(10 BYTE),
  OWNERBRANCH_ROUTING_NUMBER  NUMBER(19),
  BALANCECCY                  NUMBER(19,2),
  BALANCELCY                  NUMBER(19,2),
  NOSTROACCOUNTNUMBER         VARCHAR2(255 CHAR)
);

CREATE TABLE CONV_INSTITUTE
(
  INSTITUTETYPE  NUMBER(10),
  BANKNAME       VARCHAR2(255 CHAR),
  SHORTNAME      VARCHAR2(255 CHAR),
  COUNTRYNAME    VARCHAR2(255 CHAR),
  SWIFTCODE      VARCHAR2(255 CHAR),
  ADDRESS1       VARCHAR2(255 CHAR)
);

CREATE TABLE CONV_TREASURY_BRANCHS
(
  ADDRESS       VARCHAR2(255 CHAR),
  BRANCHCODE    NUMBER(10),
  BRANCHTYPE    VARCHAR2(255 CHAR),
  CBBRANCHCODE  VARCHAR2(255 CHAR),
  NAME          VARCHAR2(255 CHAR),
  SWIFTCODE     VARCHAR2(20 CHAR),
  CBSBRANCHID   NUMBER
);






CREATE TABLE CONV_ID_ACCOUNTS
(
   NAME                  VARCHAR2 (255),
   ACCNO                 VARCHAR2 (100),
   OPENDATE              TIMESTAMP,
   CURRENCY_ID           VARCHAR2 (10),
   BALANCECCY            NUMBER,
   BALANCELCY            NUMBER,
   NOSTROACCOUNTNUMBER   VARCHAR2 (100),
   PROD_ID               NUMBER
);
